Future<Stream<StreamEvent>> stream({
  String? path,
  PrintMode? printMode,
  FormatMode? formatMode,
  bool? shallow,
  Filter? filter,
}) async {
  final source = await client.stream(
    _buildUri(
      path: path,
      filter: filter,
      printMode: printMode,
      formatMode: formatMode,
      shallow: shallow,
    ),
    headers: _buildHeaders(),
  );
  return source.transform(const StreamEventTransformer());
}
